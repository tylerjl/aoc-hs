   17 import           Control.Monad (liftM2)
   18 import qualified Text.Parsec as     P
   19 import           Text.Parsec.Char   (digit)

   22 -- |Combinator operator for predicates
   23 (<&&>) :: (a -> Bool) -- ^ Predicate 1
   24        -> (a -> Bool) -- ^ Predicate 2
   25        -> a           -- ^ Predicate target
   26        -> Bool        -- ^ Logical and for predicates
   27 (<&&>) = liftM2 (&&)
   28 
   29 -- |Generic parsing wrapper
   30 regularParse :: Parser a -> String -> Either P.ParseError a
   31 regularParse p = P.parse p ""
   32 
   33 -- |Generic 'Int' parser
   34 intParser :: Parser Int -- ^ Parsec parser for 'Int' types
   35 intParser = read <$> P.many1 digit
