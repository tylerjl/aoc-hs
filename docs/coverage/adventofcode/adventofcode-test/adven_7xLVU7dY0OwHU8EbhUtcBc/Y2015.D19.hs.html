   10 module Y2015.D19 (distinctMols, molSteps) where
   11 
   12 import Data.Char (isLower, isUpper)
   13 import Data.List (foldl', inits, tails)
   14 import Data.Map.Strict (Map)
   15 import qualified Data.Map.Strict as M
   16 import Data.Set  (Set)
   17 import qualified Data.Set as S
   18 
   19 type Mol       = [String]
   20 type Repls     = Map String (Set String)
   21 type Compounds = Set Mol
   22 
   23 -- |Returns the number of steps required to form a specified molecule
   24 molSteps :: String -- ^ Target molecule composition as a raw string
   25          -> Int    -- ^ Number of steps required to create indicated molecule
   26 molSteps = (+) (-1) . sum . map replCount . toMol . last . lines
   27   where replCount "Rn" =  0
   28         replCount "Ar" =  0
   29         replCount "Y"  = -1
   30         replCount _    =  1
   31 
   32 -- |Finds the number of possible distinct molecules
   33 distinctMols :: String -- ^ List of starting molecules as a raw string
   34              -> Int    -- ^ Number of distinct modules that can be formed
   35 distinctMols s = S.size $ compounds mols repls
   36   where input = lines s
   37         mols  = toMol $ last input
   38         repls = toRepls $ init input
   39 
   40 compounds :: Mol -> Repls -> Set String
   41 compounds m r = foldl' S.union S.empty $ map combine molTrips
   42   where molTrips = zip3 (inits m) m (tail $ tails m)
   43         combine t@(_,m,_) = subRepl t $ M.findWithDefault S.empty m r
   44 
   45 subRepl :: (Mol, String, Mol) -> Set String -> Set String
   46 subRepl (pre,sub,post) = foldl' (flip S.insert) S.empty
   47                        . map (concat . construct) . S.toList
   48   where construct repl = pre ++ [repl] ++ post
   49 
   50 toRepls :: [String] -> Repls
   51 toRepls = M.fromListWith S.union . map (molPair S.singleton . words)
   52 
   53 molPair :: (a -> b) -> [a] -> (a, b)
   54 molPair f [from,_,to] = (from, f to)
   55 
   56 toMol :: String -> Mol
   57 toMol []                                = []
   58 toMol (x:y:ys) | isUpper x && isLower y = (x:[y]) : toMol ys
   59 toMol (x:xs)   | isUpper x || x == 'e'  = [x]     : toMol xs
   60                | otherwise              = toMol xs
