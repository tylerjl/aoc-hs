   26 data Boss = Boss
   27     { _damage :: Int
   28     , _hp     :: Int
   29     } deriving (Eq, Show)
   30 
   31 data Player = Player
   32     { _armor :: Int
   33     , _life  :: Int
   34     , _mana  :: Int
   35     , _spent :: Int
   36     } deriving (Eq, Show)
   37 
   38 data State = PlayerTurn
   39            | BossTurn
   40            deriving (Show)
   41 
   42 data Game = Game
   43     { _boss    :: Boss
   44     , _effects :: Map Spell Int
   45     , _hard    :: Bool
   46     , _player  :: Player
   47     , _state   :: State
   48     } deriving (Show)
   49 
   50 data Spell = MagicMissile
   51            | Drain
   52            | Shield
   53            | Poison
   54            | Recharge
   55            deriving (Enum, Eq, Ord, Show)
   56 
   57 -- |Represents the final result of a sequence of player moves.
   58 data Result = Won Int -- ^ Indicates a player win with the total mana spent.
   59             | Lost    -- ^ Player loss.
   60             deriving (Eq, Ord, Show)
   61 
   62 makeLenses ''Boss
   63 makeLenses ''Game
   64 makeLenses ''Player
   65 
   66 cast :: Spell -> Game -> Game
   67 cast spell = action
   68            . (player.mana -~ cost)
   69            . (player.spent +~ cost)
   70     where
   71         (action, cost) = case spell of
   72             MagicMissile -> (boss.hp -~ 4, 53)
   73             Drain        -> ((boss.hp -~ 2) . (player.life +~ 2), 73)
   74             Shield       -> (e Shield 6, 113)
   75             Poison       -> (e Poison 6, 173)
   76             Recharge     -> (e Recharge 5, 229)
   77         e s t = effects %~ insert s t
   78 
   79 affect :: Spell -> Game -> Game
   80 affect Shield   = player.armor .~ 7
   81 affect Poison   = boss.hp -~ 3
   82 affect Recharge = player.mana +~ 101
   83 affect _        = id
   84 
   85 stepEffects :: Game -> Game
   86 stepEffects game =
   87     foldr affect (game' & effects %~ step) (keys $ game^.effects)
   88     where game' = game & player.armor .~ 0
   89           step  = M.filter (0 <) . M.map pred
   90 
   91 stepGame :: Game -> [Result]
   92 stepGame game
   93     | game^.boss.hp <= 0 =
   94         [Won $ game^.player.spent]
   95     | game^.player.life <= 0 || game^.player.mana < 0 =
   96         [Lost]
   97     | otherwise =
   98         case game^.state of
   99             BossTurn   -> stepGame $ strike game' & state .~ PlayerTurn
  100             PlayerTurn -> stepSpells game'
  101             where game' = stepEffects $ case (game^.hard, game^.state) of
  102                               (True, PlayerTurn) -> game & player.life -~ 1
  103                               _ -> game
  104 
  105 stepSpells :: Game -> [Result]
  106 stepSpells game =
  107     [ result | spell <-
  108         [ s | s <- [MagicMissile ..]
  109             , not $ member s $ game^.effects
  110             ]
  111         , let nextTurn = cast spell game & state .~ BossTurn
  112         , result <- stepGame nextTurn
  113     ]
  114 
  115 strike :: Game -> Game
  116 strike g =
  117     player.life -~ max 1 (g^.boss.damage - g^.player.armor) $ g
  118 
  119 newGame :: Bool -> String -> Game
  120 newGame hardMode input =
  121     Game
  122         { _player  = Player
  123             { _life = 50
  124             , _armor = 0
  125             , _mana = 500
  126             , _spent = 0
  127             }
  128         , _boss    = boss
  129         , _effects = M.empty
  130         , _state   = PlayerTurn
  131         , _hard    = hardMode
  132         }
  133     where boss = pBoss input
  134 
  135 pBoss :: String -> Boss
  136 pBoss input = Boss { _hp = hp, _damage = dmg }
  137     where parse f = read $ last $ words $ f $ lines input
  138           hp      = parse head
  139           dmg     = parse last
  140 
  141 -- |Finds the minimum required mana to win a game.
  142 spellBattle :: Bool   -- ^ Whether to run the game in "hard mode"
  143             -> String -- ^ Boss stats input string
  144             -> Result -- ^ Best possible battle outcome
  145 spellBattle hardMode = minimum . stepGame . newGame hardMode
  146 
  147 -- |Provided as a quicker method for testing shorter player battles.
  148 testSpellBattle :: Bool   -- ^ Whether to run the game in "hard mode"
  149                 -> String -- ^ Boss stats input string
  150                 -> Result -- ^ Best possible battle outcome
  151 testSpellBattle d = minimum . stepGame
  152                   . (player.mana .~ 250) . (player.life .~ 10)
  153                   . newGame d
