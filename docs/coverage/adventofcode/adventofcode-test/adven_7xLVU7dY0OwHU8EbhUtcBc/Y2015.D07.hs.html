   41 type Wire = String
   42 
   43 data Atom = Val Word16 | Var String
   44           deriving (Show)
   45 
   46 data Gate = Singleton Atom
   47           | And       Atom Atom
   48           | Or        Atom Atom
   49           | LShift    Atom Int
   50           | RShift    Atom Int
   51           | Not       Atom
   52           deriving (Show)
   53 
   54 data Instruction = Instruction Gate Wire
   55                  deriving (Show)
   56 
   57 -- |Parsec parser for list of 'Instruction's
   58 circuitParser :: Parser [Instruction] -- ^ Parser
   59 circuitParser = many (pInstruction <* optional endOfLine)
   60 
   61 pInstruction :: Parser Instruction
   62 pInstruction = Instruction <$> pGate <*> pWire
   63 
   64 pGate :: Parser Gate
   65 pGate =  try (Singleton <$> atom <* lookAhead arrow)
   66      <|> try (And       <$> atom <* gate "AND"    <*> atom)
   67      <|> try (Or        <$> atom <* gate "OR"     <*> atom)
   68      <|> try (LShift    <$> atom <* gate "LSHIFT" <*> bits)
   69      <|> try (RShift    <$> atom <* gate "RSHIFT" <*> bits)
   70      <|> try (Not       <$          gate "NOT"    <*> atom)
   71 
   72 pWire :: Parser Wire
   73 pWire = arrow *> many1 letter
   74 
   75 gate :: String -> Parser ()
   76 gate s = skipMany space *> string s *> skipMany1 space
   77 
   78 bits :: Parser Int
   79 bits = intParser
   80 
   81 atom :: Parser Atom
   82 atom =  try (Var <$> many1 letter)
   83     <|> try (Val <$> read <$> many1 digit)
   84 
   85 arrow :: Parser ()
   86 arrow = skipMany1 space *> string "->" *> skipMany1 space
   87 
   88 voltageOn :: Map String Gate -> String -> Word16
   89 voltageOn m = resolve
   90     where eval :: String -> Word16
   91           eval wire = case M.lookup wire m of
   92                       Just (Singleton x) -> atom x
   93                       Just (And x y)     -> atom x .&. atom y
   94                       Just (Or x y)      -> atom x .|. atom y
   95                       Just (LShift x i)  -> shift (atom x) i
   96                       Just (RShift x i)  -> shift (atom x) (-i)
   97                       Just (Not x)       -> complement (atom x)
   98                       Nothing            -> 0
   99           resolve = memoize eval
  100           atom (Val i) = i
  101           atom (Var v) = resolve v
  102 
  103 -- |Constructs then returns resulting voltage from wiring spec
  104 wire :: String        -- ^ Wire to find voltage on
  105      -> [Instruction] -- ^ List of instructions
  106      -> Word16        -- ^ Resulting voltage on indicated wire
  107 wire s = flip voltageOn s . M.fromList . map toPair
  108     where toPair (Instruction g w) = (w, g)
  109 
  110 -- |Helper function to parse 'Instruction's
  111 parseCircuits :: String                            -- ^ Input string
  112               -> Either ParseError [Instruction] -- ^ Either parse error or 'Instruction's
  113 parseCircuits = regularParse circuitParser
  114 
  115 -- |Inject a manual instruction.
  116 override :: Word16      -- ^ Value to inject into 'Instruction'.
  117          -> Instruction -- ^ Resulting 'Instruction'.
  118 override s = Instruction (Singleton (Val s)) "b"
