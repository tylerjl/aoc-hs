   18 import Y2015.Util ((<&&>))
   19 
   20 import Data.List (group, isInfixOf)
   21 
   22 -- |Whether a given string is "nice" per the spec.
   23 isNicer :: String -- ^ Test input string
   24         -> Bool   -- ^ Whether string is nice
   25 isNicer = repeatedPair <&&> repeatedBetween
   26 
   27 repeatedPair :: String -> Bool
   28 repeatedPair []                               = False
   29 repeatedPair [x]                              = False
   30 repeatedPair (x:y:zs) | [x, y] `isInfixOf` zs = True
   31                       | otherwise             = repeatedPair (y : zs)
   32 
   33 repeatedBetween :: String -> Bool
   34 repeatedBetween []                     = False
   35 repeatedBetween [x]                    = False
   36 repeatedBetween [x,y]                  = False
   37 repeatedBetween (x:y:z:zs) | x == z    = True
   38                            | otherwise = repeatedBetween (y : z : zs)
   39 
   40 -- |Predicate to determine whether a given string is "nice".
   41 isNice :: String -- ^ Test input string.
   42        -> Bool   -- ^ Whether the given input string is nice.
   43 isNice = clean <&&> thriceVoweled <&&> twiceRow
   44 
   45 -- |Predicate to determine whether a string eschews forbidden strings.
   46 clean :: String -- ^ Input string.
   47       -> Bool   -- ^ Whether the string is clean.
   48 clean = not . flip any forbiddenStrings . flip isInfixOf
   49 
   50 forbiddenStrings :: [String]
   51 forbiddenStrings = ["ab", "cd", "pq", "xy"]
   52 
   53 -- |Predicate to determine whether a given string contains two letters
   54 -- |in a row.
   55 twiceRow :: String -- ^ Input string to test.
   56          -> Bool   -- ^ Whether the given string passes the predicate.
   57 twiceRow = any ((>1) . length) . group
   58 
   59 -- |Predicate to determine whether the given string contains at least three
   60 -- |vowels.
   61 thriceVoweled :: String -- ^ Input string to test.
   62               -> Bool   -- ^ Whether the string passes the predicate.
   63 thriceVoweled = (>2) . length . filter isVowel
   64 
   65 isVowel :: Char -> Bool
   66 isVowel = flip elem "aeiou"
