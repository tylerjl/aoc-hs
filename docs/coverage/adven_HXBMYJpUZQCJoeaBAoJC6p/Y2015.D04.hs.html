    9 module Y2015.D04

   12 
   13 import Crypto.Hash.MD5 (hash)
   14 import Data.ByteString.Base16 (encode)
   15 import qualified Data.ByteString.Char8 as C
   16 import Data.Monoid ((<>))
   17 
   18 -- |Cracks password hash to retrieve original value
   19 crack
   20   :: C.ByteString -- ^ Input hash
   21   -> Int -- ^ Number of leading zeroes
   22   -> Int -- ^ Original password value
   23 crack prefix d = head $ dropWhile (not . validSuffix) [0 ..]
   24   where
   25     validSuffix = check . encode . hash . (prefix <>) . C.pack . show
   26     check = (>= d) . C.length . C.takeWhile (== '0')
