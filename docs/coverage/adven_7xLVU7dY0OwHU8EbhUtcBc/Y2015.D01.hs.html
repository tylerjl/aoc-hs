   14 import Data.List (foldl')
   15 
   16 move :: Char -> Int
   17 move c
   18   | c == '(' = 1
   19   | c == ')' = -1
   20   | otherwise = 0
   21 
   22 -- |Find final level from list of elevator movements
   23 level
   24   :: String -- ^ List of input open/close parens
   25   -> Int -- ^ Final elevator level
   26 level = foldl' (+) 0 . map move
   27 
   28 -- |Find position that arrives at level 0
   29 basement
   30   :: String -- ^ List of input open/close parens
   31   -> Maybe Int -- ^ Possible position in string that arrives at zero
   32 basement = find 0 1 . map move
   33   where
   34     find current idx (x:xs)
   35       | current + x < 0 = Just idx
   36       | otherwise = find (current + x) (idx + 1) xs
   37     find _ _ [] = Nothing
