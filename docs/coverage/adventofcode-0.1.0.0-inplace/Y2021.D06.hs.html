   38 -- |Solution algorithm using super aggressive mutable unboxed vectors.
   39 solve6MV :: Int -> [Int] -> Int
   40 solve6MV days fish = runST $ do
   41   vec <- V.replicate 9 0
   42   forM_ fish $ \fish' -> do
   43     V.modify vec (+ 1) fish'
   44   replicateM_ days $ do
   45     newFish <- V.read vec 0
   46     V.move (V.slice 0 8 vec) (V.slice 1 8 vec)
   47     V.write vec 8 newFish
   48     V.modify vec (+ newFish) 6
   49   V.foldl' (+) 0 vec

   51 -- |Iterate for a given number of days given a starting value and return the
   52 -- total.
   53 solve6 :: Int -> [Int] -> Int
   54 solve6 n
   55   = M.foldl' (+) 0
   56   . flip (!!) n
   57   . iterate breed
   58   . M.fromListWith (+)
   59   . flip zip (repeat 1)
   60 
   61 -- |Simulate one day passing for the collection of fish.
   62 breed :: Num a => M.IntMap a -> M.IntMap a
   63 breed fish =
   64   M.insert 8 newFish $
   65   M.mapKeysWith (+) step fish
   66   where
   67     step (pred -> age) | age < 0   = 6
   68                        | otherwise = age
   69     newFish = M.findWithDefault 0 0 fish

   71 -- -- |Parse puzzle input into a list of `Int`s.
   72 parseFish :: Text -> [Int]
   73 parseFish = fromRight . regularParse' fishParser
   74   where fishParser = intParser' `sepBy1` char ',' <* newline <* eof
