   21 -- |Fancy data type to represent _game_ state, not just a card
   22 newtype Bingo a = Bingo a deriving Show

   55 -- |Transform a bingo game from multidimensional-array based to a set-based
   56 -- game.
   57 intoSet :: Card -> Card
   58 intoSet (CardArr (map catMaybes -> rows)) = CardSet rows' cols'
   59   where rows' = S.fromList $ map S.fromList rows
   60         cols' = S.fromList $ map S.fromList (transpose rows)
   61 intoSet c@(CardSet _ _) = c
   62 
   63 -- |Fortunately both A and B are just asking slightly different questions, so we
   64 -- have a higher-order function to determine how to pull the matching value from
   65 -- our resultant list.
   66 solve4 :: ([(Int, Bingo Card)] -> (Int, Bingo Card)) -> [Int] -> [Bingo Card] -> Int
   67 solve4 f x = tally . f . iterateMap x
   68 
   69 -- |Problem-defined method of scoring a card. The "in-progress" scoring isn't
   70 -- actually ever used. Probably a bug.
   71 tally :: (Int, Bingo Card) -> Int
   72 tally (n, Bingo card)  = n * sumCard card
   73 
   74 -- |Yank out the total values for a card.
   75 sumCard :: Card -> Int
   76 sumCard (CardArr board) = (getSum . mconcat . catMaybes . concat) board
   77 sumCard (CardSet rows cols)
   78   | any S.null rows = sumSet rows
   79   | otherwise = sumSet cols
   80   where sumSet = getSum . S.fold mappend (Sum 0) . S.map (S.fold mappend (Sum 0))
   81 
   82 -- |Our main recursive loop to walk through the "called" numbers. We return a
   83 -- list solely of elements that meet the "ended" predicate.
   84 iterateMap :: [Int] -> [Bingo Card] -> [(Int, Bingo Card)]
   85 iterateMap [] _ = []
   86 iterateMap (n:nums) cards = zip (repeat n) needles ++ iterateMap nums haystacks
   87   where (needles, haystacks) = mark ([], []) n cards
   88 
   89 -- |Accept a list of `Bingo Card`s and update each with the called number.
   90 mark :: ([Bingo Card], [Bingo Card]) -> Int -> [Bingo Card] -> ([Bingo Card], [Bingo Card])
   91 mark acc _ [] = acc
   92 mark (l, r) num (card:cards)
   93   | gameEnd card' = mark (card' : l, r) num cards
   94   | otherwise = mark (l, card' : r) num cards
   95   where card' = markRows num card
   96 
   97 -- |Update a `Card` - we make the determination to "end" a game here, so as soon
   98 -- as its "won" the type swaps over to the ended value.
   99 markRows :: Int -> Bingo Card -> Bingo Card
  100 markRows n (Bingo (CardArr board)) = Bingo $ CardArr $ map (map (markRow n)) board
  101   where
  102     markRow _ Nothing = Nothing
  103     markRow n' a@(Just (Sum b)) | b == n'   = Nothing
  104                                 | otherwise = a
  105 markRows n (Bingo (CardSet rows cols)) = Bingo $ CardSet (markSet rows) (markSet cols)
  106   where markSet = S.map (S.delete (Sum n))
  107 
  108 -- |Important function to determine if a card is a winning card.
  109 gameEnd :: Bingo Card -> Bool
  110 gameEnd (Bingo (CardArr card))
  111   = check card || (check . transpose) card
  112   where check = any (all isNothing)
  113 gameEnd (Bingo (CardSet rows cols))
  114   = any S.null rows || any S.null cols

  116 -- |This is a sort of hairy, but all-in-one, parser for the problem set input. 5
  117 -- is a magic number for board size.
  118 bingoParser :: Parser ([Int], [Bingo Card])
  119 bingoParser = (,) <$> (lottoParser <* newline) <*> cardsParser <* eof
  120   where
  121     lottoParser = intParser' `sepBy1` char ',' <* newline
  122     rowParser   = (Just . Sum <$> intParser') `sepBy1` many1 (char ' ')
  123     cardParser  = Bingo . CardArr <$> count 5 (optional (many space) *> rowParser <* endOfLine)
  124     cardsParser = cardParser `sepBy1` newline
