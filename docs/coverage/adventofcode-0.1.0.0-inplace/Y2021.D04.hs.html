   17 -- |Fancy data type to represent _game_ state, not just a card
   18 data Bingo a
   19   = Ended (Int, a)
   20   | Playing a
   21 -- |I'm not sure if a derive would get this right
   22 instance Functor Bingo where
   23   fmap f (Ended (i, c)) = Ended (i, f c)
   24   fmap f (Playing a) = Playing (f a)
   25 -- |Represents a game board state
   26 type Card = [[Square]]
   27 -- |Small wrapper over how to record marked/unmarked squares
   28 data Square = Marked | Unmarked Int deriving Show

   30 -- |Solve part A
   31 part4A :: Text -> Int
   32 part4A (regularParse' bingoParser -> Right (ns, cs)) = solve4 head ns cs
   33 part4A (regularParse' bingoParser -> Left (show -> err)) = error err
   34 
   35 -- |Solve part B
   36 part4B :: Text -> Int
   37 part4B (regularParse' bingoParser -> Right (ns, cs)) = solve4 last ns cs
   38 part4B (regularParse' bingoParser -> Left (show -> err)) = error err

   40 -- |Fortunately both A and B are just asking slightly different questions, so we
   41 -- have a higher-order function to determine how to pull the matching value from
   42 -- our resultant list.
   43 solve4 :: ([Bingo Card] -> Bingo Card) -> [Int] -> [Bingo Card] -> Int
   44 solve4 f x = tally . f . iterateMap x
   45 
   46 -- |Problem-defined method of scoring a card. The "in-progress" scoring isn't
   47 -- actually ever used. Probably a bug.
   48 tally :: Bingo Card -> Int
   49 tally (Playing card)  = sumCard card
   50 tally (Ended (n, card)) = n * sumCard card
   51 
   52 -- |Yank out the total values for a card.
   53 sumCard :: Card -> Int
   54 sumCard = sum . map (\(Unmarked n') -> n') . filter (not . marked) . concat
   55 
   56 -- |Our main recursive loop to walk through the "called" numbers. We return a
   57 -- list solely of elements that meet the "ended" predicate.
   58 iterateMap :: [Int] -> [Bingo Card] -> [Bingo Card]
   59 iterateMap [] _ = []
   60 iterateMap (n:nums) cards = needles ++ iterateMap nums haystacks
   61   where (needles, haystacks) = partition hasEnded $ mark n cards
   62 
   63 -- |Accept a list of `Bingo Card`s and update each with the called number.
   64 mark :: Int -> [Bingo Card] -> [Bingo Card]
   65 mark num = map (markRows num)
   66 
   67 -- |Update a `Card` - we make the determination to "end" a game here, so as soon
   68 -- as its "won" the type swaps over to the ended value.
   69 markRows :: Int -> Bingo Card -> Bingo Card
   70 markRows n = gameEnd n . fmap (fmap $ map (markRow n))
   71   where
   72     markRow _ Marked = Marked
   73     markRow n' a@(Unmarked b) | b == n'   = Marked
   74                               | otherwise = a
   75 
   76 -- |Small helper predicate to tell is when a `Bingo` type is over or not.
   77 hasEnded :: Bingo a -> Bool
   78 hasEnded (Playing _) = False
   79 hasEnded (Ended _) = True
   80 
   81 -- |Important function to determine if a card is a winning card. If so it
   82 -- changes its type.
   83 gameEnd :: Int -> Bingo Card -> Bingo Card
   84 gameEnd _ b@(Ended _) = b
   85 gameEnd n b@(Playing card)
   86   | check card || (check . transpose) card = Ended (n, card)
   87   | otherwise = b
   88   where check = any (all marked)
   89 
   90 -- |Another little helper to tell us if a spot has been marked or not.
   91 marked :: Square -> Bool
   92 marked Marked       = True
   93 marked (Unmarked _) = False
   94 
   95 -- |This is a sort of hairy, but all-in-one, parser for the problem set input. 5
   96 -- is a magic number for board size.
   97 bingoParser :: Parser ([Int], [Bingo Card])
   98 bingoParser = (,) <$> (lottoParser <* newline) <*> cardsParser <* eof
   99   where
  100     lottoParser = intParser' `sepBy1` char ',' <* newline
  101     rowParser   = (Unmarked <$> intParser') `sepBy1` many1 (char ' ')
  102     cardParser  = Playing <$> count 5 (optional (many space) *> rowParser <* endOfLine)
  103     cardsParser = cardParser `sepBy1` newline
