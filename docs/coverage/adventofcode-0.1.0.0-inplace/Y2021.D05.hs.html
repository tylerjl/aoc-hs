   39 part5APredicate :: (Eq a1, Eq a2) => ((a1, a2), (a1, a2)) -> Bool
   40 part5APredicate ((x1, y1), (x2, y2))
   41   = x1 == x2 || y1 == y2

   43 -- |Solve part B
   44 part5B :: Text -> Int
   45 part5B = solve5 id

   48 part5BHM :: Text -> Int

   51 -- |Solve part B, but with a MultiSet
   52 part5BMS :: Text -> Int
   53 part5BMS =
   54   length . filter ((> 1) . snd) . MS.toOccurList . fromRight . parseOnly parser
   55   where
   56     parser = foldl' MS.union MS.empty <$> lines'
   57     lines' = line `sepBy1` endOfLine <* atEnd
   58     line = MS.fromList . uncurry lineTo' <$> toFrom
   59     toFrom  = (,) <$> point <* string " -> " <*> point
   60     point = (,) <$> decimal <* char ',' <*> decimal
   61 
   62 -- |Higher-order function solution to share parts A and B.
   63 solve5 :: (Ray -> Ray) -> Text -> Int
   64 solve5 f =
   65   M.size . M.filter (> 1) . M.fromListWith mappend .
   66   concatMap (uncurry lineTo) . f . parseVents
   67 
   68 -- |Same as `solve5`, but using HashMap.
   69 solve5HM :: (Ray -> Ray) -> Text -> Int
   70 solve5HM f =
   71   HM.size . HM.filter (> 1) . HM.fromListWith mappend .
   72   concatMap (uncurry lineTo) . f . parseVents
   73 
   74 -- |Same as `solve5`, but using MultiSet
   75 solve5MS :: (Ray -> Ray) -> Text -> Int
   76 solve5MS f =
   77   length . filter ((> 1) . snd) . MS.toOccurList . MS.fromList .
   78   concatMap (uncurry lineTo') . f . parseVents

   80 -- |Accept a start and end point and return a list of points that draw a line to
   81 -- the endpoint. Note that this doesn't work for anything other than vertical,
   82 -- horizontal, and 45deg.
   83 lineTo :: Point -> Point -> [(Point, Sum Int)]
   84 lineTo (x1, y1) (x2, y2) = zipWith (curry (, Sum 1)) (range x1 x2) (range y1 y2)
   85   where range p1 p2 | p1 == p2  = repeat p1
   86                     | otherwise = p1 ~~ p2
   87 
   88 -- |Same as `lineTo` but without the overhead for a MultiSet
   89 lineTo' :: Point -> Point -> [Point]
   90 lineTo' (x1, y1) (x2, y2) = zip (range x1 x2) (range y1 y2)
   91   where range p1 p2 | p1 == p2  = repeat p1
   92                     | otherwise = p1 ~~ p2
   93 
   94 -- |Messing around with a custom operator for ranges that can handle both up-to
   95 -- and down-to.
   96 infixl 5 ~~
   97 (~~) :: Int -> Int -> [Int]
   98 a ~~ b | a <= b    = [a .. b]
   99        | otherwise = reverse $ b ~~ a

  101 -- |Parse puzzle input into simple pairs of pairs of points.
  102 parseVents :: Text -> Ray
  103 parseVents = fromRight . parseOnly parser
  104   where
  105     parser = line `sepBy1` endOfLine <* atEnd
  106     line  = (,) <$> point <* string " -> " <*> point
  107     point = (,) <$> decimal <* char ',' <*> decimal
