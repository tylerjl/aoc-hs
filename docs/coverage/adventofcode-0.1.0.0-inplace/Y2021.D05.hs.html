   21 -- |Makes some signatures easier to read
   22 type Ray = [(Point, Point)]

   24 -- |Solve part A
   25 part5A :: Text -> Int
   26 part5A = solve5 (filter noAngle)
   27   where noAngle ((x1, y1), (x2, y2))
   28           = x1 == x2 || y1 == y2
   29 
   30 -- |Solve part B
   31 part5B :: Text -> Int
   32 part5B = solve5 id
   33 
   34 -- |Higher-order function solution to share parts A and B.
   35 solve5 :: (Ray -> Ray) -> Text -> Int
   36 solve5 f =
   37   M.size . M.filter (> (1 :: Int)) . M.fromListWith (+) .
   38   concatMap (map (, 1) . uncurry lineTo) . f . parseVents
   39 
   40 -- |Accept a start and end point and return a list of points that draw a line to
   41 -- the endpoint. Note that this doesn't work for anything other than vertical,
   42 -- horizontal, and 45deg.
   43 lineTo :: Point -> Point -> [Point]
   44 lineTo (x1, y1) (x2, y2) = zip (range x1 x2) (range y1 y2)
   45   where range p1 p2 | p1 == p2  = repeat p1
   46                     | otherwise = p1 ~~ p2
   47 
   48 -- |Messing around with a custom operator for ranges that can handle both up-to
   49 -- and down-to.
   50 (~~) :: Int -> Int -> [Int]
   51 a ~~ b | a <= b    = [a .. b]
   52        | otherwise = reverse $ b ~~ a
   53 
   54 -- |Should be fun!
   55 infixl 5 ~~

   57 -- |Parse puzzle input into simple pairs of pairs of points.
   58 parseVents :: Text -> Ray
   59 parseVents = fromRight . regularParse' (line `endBy1` newline <* eof)
   60   where
   61     line  = (,) <$> point <* string " -> " <*> point
   62     point = (,) <$> intParser' <* char ',' <*> intParser'
