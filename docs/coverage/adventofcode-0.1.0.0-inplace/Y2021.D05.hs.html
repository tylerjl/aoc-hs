   27 -- |Solve part A
   28 part5A :: Text -> Int
   29 part5A = solve5 (filter part5APredicate)
   30 
   31 part5APredicate :: Line -> Bool
   32 part5APredicate ((x1, y1), (x2, y2))
   33   = x1 == x2 || y1 == y2
   34 
   35 -- |Solve part B
   36 part5B :: Text -> Int
   37 part5B = solve5 id
   38 
   39 -- |Higher-order function solution to share parts A and B.
   40 solve5 :: ([Line] -> [Line]) -> Text -> Int
   41 solve5 f =
   42   M.size .
   43   M.filter (> 1) . M.fromListWith mappend . concatMap lineTo . f . parseVents
   44 
   45 -- |Accept a start and end point and return a list of points that draw a line to
   46 -- the endpoint. Note that this doesn't work for anything other than vertical,
   47 -- horizontal, and 45deg.
   48 lineTo :: Line -> [(Point, Sum Int)]
   49 lineTo ((x1, y1), (x2, y2)) = zipWith (curry (, Sum 1)) (range x1 x2) (range y1 y2)
   50   where range p1 p2 | p1 == p2  = repeat p1
   51                     | otherwise = p1 ~~ p2
   52 
   53 -- |Messing around with a custom operator for ranges that can handle both up-to
   54 -- and down-to.
   55 infixl 5 ~~
   56 (~~) :: Int -> Int -> [Int]
   57 a ~~ b | a <= b    = [a .. b]
   58        | otherwise = reverse $ b ~~ a
   59 
   60 -- |Parse puzzle input into simple pairs of pairs of points.
   61 parseVents :: Text -> [Line]
   62 parseVents = fromRight . parseOnly parser
   63   where
   64     parser = line `sepBy1` endOfLine <* atEnd
   65     line  = (,) <$> point <* string " -> " <*> point
   66     point = (,) <$> decimal <* char ',' <*> decimal
