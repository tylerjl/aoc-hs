   26 -- |Solve part A
   27 part5A :: Text -> Int
   28 part5A = solve5 (filter noAngle)
   29   where noAngle ((x1, y1), (x2, y2))
   30           = x1 == x2 || y1 == y2
   31 
   32 -- |Solve part A, but with a HashMap
   33 part5AHM :: Text -> Int
   34 part5AHM = solve5HM (filter noAngle)
   35   where noAngle ((x1, y1), (x2, y2))
   36           = x1 == x2 || y1 == y2
   37 
   38 -- |Solve part B
   39 part5B :: Text -> Int
   40 part5B = solve5 id
   41 
   42 -- |Solve part B, but with a HashMap
   43 part5BHM :: Text -> Int
   44 part5BHM = solve5HM id

   46 -- |Higher-order function solution to share parts A and B.
   47 solve5 :: (Ray -> Ray) -> Text -> Int
   48 solve5 f =
   49   M.size . M.filter (> 1) . M.fromListWith mappend .
   50   concatMap (uncurry lineTo) . f . parseVents
   51 
   52 -- |Same as `solve5`, but using HashMap.
   53 solve5HM :: (Ray -> Ray) -> Text -> Int
   54 solve5HM f =
   55   HM.size . HM.filter (> 1) . HM.fromListWith mappend .
   56   concatMap (uncurry lineTo) . f . parseVents
   57 
   58 -- |Accept a start and end point and return a list of points that draw a line to
   59 -- the endpoint. Note that this doesn't work for anything other than vertical,
   60 -- horizontal, and 45deg.
   61 lineTo :: Point -> Point -> [(Point, Sum Int)]
   62 lineTo (x1, y1) (x2, y2) = zipWith (curry (, Sum 1)) (range x1 x2) (range y1 y2)
   63   where range p1 p2 | p1 == p2  = repeat p1
   64                     | otherwise = p1 ~~ p2
   65 
   66 -- |Messing around with a custom operator for ranges that can handle both up-to
   67 -- and down-to.
   68 infixl 5 ~~
   69 (~~) :: Int -> Int -> [Int]
   70 a ~~ b | a <= b    = [a .. b]
   71        | otherwise = reverse $ b ~~ a
   72 
   73 -- |Parse puzzle input into simple pairs of pairs of points.
   74 parseVents :: Text -> Ray
   75 parseVents = fromRight . regularParse' (line `endBy1` newline <* eof)
   76   where
   77     line  = (,) <$> point <* string " -> " <*> point
   78     point = (,) <$> intParser' <* char ',' <*> intParser'
