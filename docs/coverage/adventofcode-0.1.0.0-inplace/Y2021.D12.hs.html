   30 -- |This GADT helps quite a bit when comparing vaules later on
   31 data Room = Start | BigCave Text | SmallCave Text | End
   32   deriving (Eq, Ord, Show)
   33 -- |Type alias for better readability.
   34 type Caves = Map Room (Set Room)
   35 -- |Type alias for better readability.
   36 type Journey = [Room]
   37 
   38 -- |Solution to part A
   39 part12A :: Text -> Int
   40 part12A (parse12' -> rooms)
   41   = length $ explore smallOnce rooms mempty Start
   42 
   43 -- |This is the main recursive function that drives both A and B. The
   44 -- higher-order function argument is the main difference, bit otherwise we build
   45 -- up a list of all valid routes by traversing the overall `Map`.
   46 explore :: (Journey -> Room -> Bool) -> Caves -> Journey -> Room -> [Journey]
   47 explore allowed caves journey room
   48   | room == End = [visited]
   49   | otherwise = foldMap (concatMap (explore allowed caves visited)) rooms
   50   where
   51     visited = room : journey
   52     rooms = S.filter (allowed visited) <$> M.lookup room caves
   53 
   54 -- |The predicate we use for part A, which is essentially "don't visit small
   55 -- rooms twice"
   56 smallOnce :: Journey -> Room -> Bool
   57 smallOnce journey c@(SmallCave _) = c `notElem` journey
   58 smallOnce _ Start = False
   59 smallOnce _ _ = True
   60 
   61 -- |Solution to part B
   62 part12B :: Text -> Int
   63 part12B (parse12' -> rooms)
   64   = length $ explore smallTwice rooms mempty Start
   65 
   66 -- |Given our journey through the caves so far and the room we'd like to proceed
   67 -- through, should we continue?
   68 --
   69 -- Part B requires a little more logic, which is primarily to visit any single
   70 -- small cave at most two times.
   71 smallTwice :: Journey -> Room -> Bool
   72 smallTwice _ (BigCave _) = True
   73 smallTwice _ Start = False
   74 smallTwice _ End = True
   75 smallTwice (filter isSmall -> j) r
   76   | isJust seenTwice = r `notElem` j
   77   | otherwise = True
   78   where
   79     seenTwice = find ((==) 2 . flip occursInList j) j
   80 
   81 -- |Super small utility to find small caves.
   82 isSmall :: Room -> Bool
   83 isSmall (SmallCave _) = True
   84 isSmall _ = False
   85 
   86 -- |Generic function to count occurrences in a list. This is probably pretty
   87 -- inefficient.
   88 occursInList :: Eq a => a -> [a] -> Int
   89 occursInList x = length . filter (x ==)

   91 -- |An intermediate parsing function; once we get the raw room pairs we turn it
   92 -- into the structure we'll work with later.
   93 parse12' :: Text -> Map Room (Set Room)
   94 parse12' =
   95   M.fromListWith S.union .
   96   map (second S.singleton) . nub . concatMap backpath . parse12
   97   where
   98     backpath (a, b) = [(a, b), (b, a)]
   99 
  100 -- |Pairs of cave "rooms", which we parse very simply and postprocess later.
  101 parse12 :: Text -> [(Room, Room)]
  102 parse12 = fromRight . parseOnly parser
  103   where
  104     parser    = line `sepBy1` endOfLine <* atEnd
  105     line      = (,) <$> location <* char '-' <*> location
  106     location  = start' <|> end' <|> bigCave <|> smallCave
  107     start'    = Start                  <$  string "start"
  108     bigCave   = BigCave   . into @Text <$> many1 (satisfy (`elem` ['A'..'Z']))
  109     smallCave = SmallCave . into @Text <$> many1 (satisfy (`elem` ['a'..'z']))
  110     end'      = End                    <$  string "end"
