   13 import Data.Text (Text)
   14 import qualified Text.Parsec as P
   15 import Text.Parsec.Text (Parser)
   16 import Y2015.Util (regularParse', intParser')
   17 import Control.Applicative ((<|>))
   18 
   19 data Navigation
   20   = Forward Int
   21   | Down Int
   22   | Up Int
   23 
   24 navParser :: Parser [Navigation]
   25 navParser = P.many (parseNav <* P.optional P.endOfLine)
   26 
   27 parseNav :: Parser Navigation
   28 parseNav = Forward <$> pInstr "forward"
   29        <|> Down    <$> pInstr "down"
   30        <|> Up      <$> pInstr "up"
   31 
   32 pInstr :: String -> Parser Int
   33 pInstr s = P.string s *> P.skipMany1 P.space *> intParser'
   34 
   35 part2A :: Text -> Int
   36 part2A (regularParse' navParser -> Right navs) = uncurry (*) $ go (0, 0) navs
   37   where go (x, y) (Forward n:xs) = go (x + n, y) xs
   38         go (x, y) (Down n:xs)    = go (x, y + n) xs
   39         go (x, y) (Up n:xs)      = go (x, y - n) xs
   40         go coords []             = coords
   41 part2A (regularParse' navParser -> Left err) = error (show err)
   42 
   43 part2B :: Text -> Int
   44 part2B (regularParse' navParser -> Right navs) = (\(x, y, _) -> x * y) $ go (0, 0, 0) navs
   45   where go (x, y, aim) (Forward n:xs) = go (x + n, y + (aim * n), aim) xs
   46         go (x, y, aim) (Down n:xs)    = go (x, y, aim + n) xs
   47         go (x, y, aim) (Up n:xs)      = go (x, y, aim - n) xs
   48         go coords []                  = coords
   49 part2B (regularParse' navParser -> Left err) = error (show err)
