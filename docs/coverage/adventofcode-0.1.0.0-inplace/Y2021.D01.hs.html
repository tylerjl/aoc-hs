   25 -- |Part a using simple recursion (maybe smaller big-O?)
   26 partARecur :: Text -> Int
   27 partARecur = go . asInts

   39 -- |Utility to transform a list into tuples of adjacent values.
   40 compareAdj :: [Int] -> [(Int, Int)]
   41 compareAdj = filter (uncurry (<)) . (zip <*> tail)
   42 
   43 -- |Sum all values of a three-tuple
   44 trisum :: Num a => (a, a, a) -> a
   45 trisum (a, b, c) = a + b + c
   46 
   47 -- |A fold sum of the structure for parsed values.
   48 stepwise :: [Int] -> Int
   49 stepwise = snd . foldl' steps (Nothing, 0)
   50   where steps :: (Maybe Int, Int) -> Int -> (Maybe Int, Int)
   51         steps (Nothing,   acc) n = (Just n, acc)
   52         steps (Just prev, acc) n
   53           | prev < n  = (Just n, succ acc)
   54           | otherwise = (Just n, acc)
   55 
   56 -- |Consolidate a list into a summed, shifting window.
   57 toWindows :: [Int] -> [Int]
   58 toWindows (w:x:y:zs) = (w+x+y) : toWindows (x:y:zs)
   59 toWindows _ = []
   60 
   61 -- |Unsafe-ish `Text` traversal to transform into a list of `Int`s.
   62 asInts :: Text -> [Int]
   63 asInts = map (read . into @String) . T.lines
