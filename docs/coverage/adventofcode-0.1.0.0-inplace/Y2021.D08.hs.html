   29 -- |Guess I'll use a sum type for this
   30 data Signal = SigA | SigB | SigC | SigD | SigE | SigF | SigG
   31   deriving (Bounded, Enum, Generic, Hashable, Eq, Ord, Show, NFData)

   33 -- |These are just for readability
   34 type SignalSequence = [Signal]
   35 type SignalSet = Set Signal
   36 type SignalEntries = [([SignalSequence], [SignalSequence])]

   47 solve8A :: [(a, [SignalSequence])] -> Int

   50 -- |Our simple predicate for part A per the instructions.
   51 part8APredicate :: SignalSequence -> Bool

   53   = size == 2 || size == 4 || size == 3 || size == 7

   59 -- |Brute-force the input to decode the output.
   60 decode :: [SignalSequence] -> [SignalSequence] -> Sum Int
   61 decode (untangle -> Just decoded) =
   62   mconcat . zipWith pow ([0..] :: [Int]) . reverse . mapMaybe decode'
   63   where decode' (S.fromList . map tr -> signals) =
   64           fromEnum <$> M.lookup signals sigMap
   65         tr sig = M.findWithDefault sig sig decoded
   66         pow n int = Sum $ (10 ^ n) * int
   67 decode (untangle -> Nothing) = error . mappend "could not decode" . show
   68 
   69 -- Come up with the right mapping between scrambled signals and their true outputs.
   70 untangle :: [SignalSequence] -> Maybe (HashMap Signal Signal)
   71 untangle (S.fromList . map S.fromList -> coded) = find mapping all'
   72   where
   73     all' = map (M.fromList . zip [ minBound .. maxBound ])
   74       $ permutations [ minBound .. maxBound ]
   75     mapping t = S.map (S.map $ replace t) coded == correctSet
   76     replace test entry = M.findWithDefault entry entry test
   77 
   78 -- |Compile the mappings for the golden input.
   79 correctSet :: Set SignalSet
   80 correctSet = S.fromList $ map (S.fromList . dMap') [minBound .. maxBound]
   81 
   82 -- |Mappings from the right signals to their associated displays.
   83 sigMap :: HashMap SignalSet Display
   84 sigMap = M.fromList $ zip (map (S.fromList . dMap') displays) displays
   85   where displays = [ minBound .. maxBound ]
   86 
   87 -- |From `Display` -> the right sequence.
   88 dMap' :: Display -> SignalSequence
   89 dMap' D0 = [SigA, SigB, SigC, SigE, SigF, SigG]
   90 dMap' D1 = [SigC, SigF]
   91 dMap' D2 = [SigA, SigC, SigD, SigE, SigG]
   92 dMap' D3 = [SigA, SigC, SigD, SigF, SigG]
   93 dMap' D4 = [SigB, SigC, SigD, SigF]
   94 dMap' D5 = [SigA, SigB, SigD, SigF, SigG]
   95 dMap' D6 = [SigA, SigB, SigD, SigE, SigF, SigG]
   96 dMap' D7 = [SigA, SigC, SigF]
   97 dMap' D8 = [SigA, SigB, SigC, SigD, SigE, SigF, SigG]
   98 dMap' D9 = [SigA, SigB, SigC, SigD, SigF, SigG]

  100 -- |Parse puzzle input into a list of `Int`s with faster attoparsec.
  101 parseSevSeg :: Text -> SignalEntries
  102 parseSevSeg = fromRight . parseOnly parser
  103   where
  104     parser = entry `sepBy1` endOfLine <* endOfLine
  105     entry = (,) <$> pattern' <* string " | " <*> pattern'
  106     pattern' = many1 signal `sepBy1` satisfy isHorizontalSpace
  107     signal = SigA <$ char 'a'
  108              <|> SigB <$ char 'b'
  109              <|> SigC <$ char 'c'
  110              <|> SigD <$ char 'd'
  111              <|> SigE <$ char 'e'
  112              <|> SigF <$ char 'f'
  113              <|> SigG <$ char 'g'
